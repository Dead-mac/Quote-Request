
import React, { useState } from 'react';
import { Vehicle, FormData } from '../types';
import { FormField, FormSection } from './FormElements';
import { vehicleFields } from './formFields';
import { CheckIcon } from './icons/CheckIcon';

type VinLookupState = {
    isLoading: boolean;
    error: string | null;
    success: boolean;
};

interface VehicleFormSectionProps {
    vehicles: Partial<Vehicle>[];
    setFormData: React.Dispatch<React.SetStateAction<any>>;
}

const VehicleFormSection: React.FC<VehicleFormSectionProps> = ({ vehicles, setFormData }) => {
    const [vinLookupStates, setVinLookupStates] = useState<Record<number, VinLookupState>>({});

    const handleVehicleChange = (
        index: number,
        e: React.ChangeEvent<HTMLInputElement | HTMLTextAreaElement>
    ) => {
        const { name, value } = e.target;
        setFormData((prev: FormData) => {
            const newVehicles = [...(prev.vehicles || [])];
            newVehicles[index] = { ...newVehicles[index], [name]: value };
            return { ...prev, vehicles: newVehicles };
        });

        // Clear VIN error and success on change for a better user experience
        if (name === 'vin') {
            setVinLookupStates(prev => {
                const newStates = { ...prev };
                if (newStates[index]) {
                  newStates[index].error = null;
                  newStates[index].success = false;
                }
                return newStates;
            });
        }
    };

    const addVehicle = () => {
        setFormData((prev: FormData) => ({
            ...prev,
            vehicles: [...(prev.vehicles || []), { id: Date.now() }],
        }));
    };
    
    const removeVehicle = (index: number) => {
        setFormData((prev: FormData) => ({
            ...prev,
            vehicles: (prev.vehicles || []).filter((_, i) => i !== index),
        }));
    };

    const handleVinLookup = async (index: number) => {
        const vin = vehicles[index]?.vin?.trim();
        if (!vin || vin.length !== 17) {
            setVinLookupStates(prev => ({ ...prev, [index]: { isLoading: false, error: 'Please enter a valid 17-character VIN.', success: false } }));
            return;
        }

        setVinLookupStates(prev => ({ ...prev, [index]: { isLoading: true, error: null, success: false } }));

        try {
            const response = await fetch(`https://vpic.nhtsa.dot.gov/api/vehicles/DecodeVinValues/${vin}?format=json`);
            if (!response.ok) {
                throw new Error('Network response was not ok.');
            }
            const data = await response.json();
            
            if (data.Results && data.Results.length > 0 && data.Results[0].ErrorCode === '0') {
                const vehicleData = data.Results[0];
                setFormData((prev: FormData) => {
                    const newVehicles = [...(prev.vehicles || [])];
                    const targetVehicle = { ...newVehicles[index] };
                    targetVehicle.year = vehicleData.ModelYear || '';
                    targetVehicle.make = vehicleData.Make || '';
                    targetVehicle.model = vehicleData.Model || '';
                    targetVehicle.trim = vehicleData.Trim || '';
                    newVehicles[index] = targetVehicle;
                    return { ...prev, vehicles: newVehicles };
                });
                setVinLookupStates(prev => ({ ...prev, [index]: { isLoading: false, error: null, success: true } }));
            } else {
                const errorMessage = data.Results[0]?.ErrorText || 'VIN not found.';
                throw new Error(errorMessage);
            }
        } catch (error: any) {
            setVinLookupStates(prev => ({ ...prev, [index]: { isLoading: false, error: error.message || 'Failed to lookup VIN.', success: false } }));
        }
    };

    return (
        <>
            {vehicles.map((vehicle, index) => (
                <div key={vehicle.id} className="md:col-span-2 contents">
                    <FormSection title={`Vehicle ${index + 1} Details`} intro="All vehicles to be insured">
                        {vehicleFields.map(field => {
                            const lookupState = vinLookupStates[index] || { isLoading: false, error: null, success: false };
                            const isSuccess = ['year', 'make', 'model', 'trim'].includes(field.name) && lookupState.success;

                            if (field.name === 'vin') {
                                return (
                                    <div key={field.name} className="md:col-span-2">
                                        <label htmlFor={`vehicle${index}.${field.name}`} className="block text-sm font-medium text-zinc-700 mb-1">
                                            {field.label}
                                        </label>
                                        <div className="flex flex-col sm:flex-row items-start sm:items-center gap-2">
                                            <div className="relative w-full">
                                                <input
                                                    type="text"
                                                    id={`vehicle${index}.${field.name}`}
                                                    name={field.name}
                                                    value={(vehicle?.[field.name] as string) || ''}
                                                    onChange={(e) => handleVehicleChange(index, e)}
                                                    className={`w-full px-4 py-3 bg-white border rounded-xl focus:outline-none focus:ring-2 transition-colors duration-200 ${
                                                        lookupState.error
                                                            ? 'border-red-500 focus:border-red-500 focus:ring-red-500/20'
                                                            : 'border-zinc-300 hover:border-zinc-400 focus:border-teal-500 focus:ring-teal-500/20'
                                                    }`}
                                                />
                                                {lookupState.success && (
                                                    <div className="absolute inset-y-0 right-0 flex items-center pr-3 pointer-events-none">
                                                        <CheckIcon className="w-5 h-5 text-green-600" />
                                                    </div>
                                                )}
                                            </div>
                                            <button
                                                type="button"
                                                onClick={() => handleVinLookup(index)}
                                                disabled={lookupState.isLoading}
                                                className="w-full sm:w-auto px-4 py-3 text-sm bg-zinc-700 text-white font-semibold rounded-lg hover:bg-zinc-800 transition-colors disabled:bg-zinc-400 disabled:cursor-not-allowed flex-shrink-0"
                                            >
                                                {lookupState.isLoading ? 'Searching...' : 'Lookup VIN'}
                                            </button>
                                        </div>
                                        {lookupState.error && <p className="text-red-600 text-xs mt-1">{lookupState.error}</p>}
                                    </div>
                                );
                            }
                            return (
                                <FormField
                                    key={field.name}
                                    id={`vehicle${index}.${field.name}`}
                                    name={field.name}
                                    label={field.label}
                                    placeholder={field.placeholder}
                                    as={field.as}
                                    value={(vehicle?.[field.name] as string) || ''}
                                    onChange={(e) => handleVehicleChange(index, e)}
                                    success={isSuccess}
                                />
                            );
                        })}
                        {vehicles.length > 1 && (
                            <div className="md:col-span-2 flex justify-end">
                                <button type="button" onClick={() => removeVehicle(index)} className="text-sm font-semibold text-red-600 hover:text-red-800">
                                    Remove Vehicle
                                </button>
                            </div>
                        )}
                    </FormSection>
                </div>
            ))}
            <div className="md:col-span-2 mt-4 flex justify-start">
                <button type="button" onClick={addVehicle} className="px-4 py-2 text-sm bg-zinc-100 text-zinc-700 font-semibold rounded-lg hover:bg-zinc-200 transition-colors">
                    + Add Another Vehicle
                </button>
            </div>
        </>
    );
};

export default VehicleFormSection;
